fn set_function_source_hash(
    &mut self,
    instance: Instance<'tcx>,
    function_source_hash: u64
) -> bool

Returns true if the function source hash was added to the coverage map (even if it had already been added, for this instance). Returns false only if -C instrument-coverage is not enabled (a coverage map is not being generated).
